package com.example.calculatormx

import android.util.Log
import androidx.lifecycle.LiveData
import androidx.lifecycle.MutableLiveData
import androidx.lifecycle.ViewModel
import org.mozilla.javascript.Context
import org.mozilla.javascript.Scriptable
import kotlin.math.log

class CalculatorViewModel : ViewModel() {
    private val _mathedText = MutableLiveData("")
    val mathedText : LiveData<String> = _mathedText

    private val _resultText = MutableLiveData("0")
    val resultText :LiveData<String> = _resultText
    fun onButttonClick(btn:String){
        Log.i("Clicked btn", btn)
        _mathedText.value?.let {
            if (btn=="AC"){
                _mathedText.value = ""
                _resultText.value = "0"
                return
            }
            if (btn== "C"){
                if (it.isNotEmpty()){
                    _mathedText.value = it.substring(0,it.length-1)
                }
                return
            }
            if (btn=="="){
                _mathedText.value = _resultText.value
                return
            }
        _mathedText.value= it+btn
            try {
                _resultText.value = calculateResult(_mathedText.value.toString())
            }catch (_ : Exception){}
        }}
        fun calculateResult(mathed : String) : String{
            val context : Context = Context.enter()
            context.optimizationLevel = -1
            val scriptable : Scriptable = context.initStandardObjects()
           var finalResult = context.evaluateString(scriptable,mathed,"Javascript",1,null).toString()
           if(finalResult.endsWith(".0")){
               finalResult = finalResult.replace(".0","")
           }

            return finalResult
        }
    }
