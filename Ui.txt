val buttonsList = listOf(
    "C","(",")","/",
    "7","8","9","*",
    "4","5","6","+",
    "1","2","3","-",
    "AC","0",".","="
)

@Composable
fun Calculator(modifier: Modifier = Modifier,viewModel: CalculatorViewModel){
    val mathedText = viewModel.mathedText.observeAsState()
    val resultText = viewModel.resultText.observeAsState()

    Box ( modifier = modifier,
        ){
        Column (modifier = modifier.fillMaxSize(),
            horizontalAlignment = Alignment.End){
            Text(text = mathedText.value?:"",
                style = TextStyle(
                    fontSize = 30.sp,
              textAlign = TextAlign.End
                ),
                maxLines = 5,
                overflow = TextOverflow.Ellipsis
            )
            Spacer(modifier = Modifier.weight(1f))
            Text(text = resultText.value?:"",
                style = TextStyle(
                    fontSize = 60.sp,
                    textAlign = TextAlign.End
                ),
                maxLines = 2,
            )
            Spacer(modifier = Modifier.height(10.dp))

            LazyVerticalGrid(columns = GridCells.Fixed(4),) {
                items(buttonsList){
             CalculateBtn(btn = it, onClick = {
                 viewModel.onButttonClick(it)
             })
                }
            }
        }

    }
}
@Composable
fun CalculateBtn(btn:String,onClick: ()-> Unit){
Box(modifier = Modifier.padding(10.dp)) {
FloatingActionButton(onClick = onClick,
    modifier = Modifier.size(80.dp),
    contentColor = Color(0xFF2990ef),
    containerColor = getColor(btn)
) {
    Text(text = btn, fontSize = 23.sp, fontWeight = FontWeight.Bold)
}
}}

    @SuppressLint("SuspiciousIndentation")
    fun getColor(btn:String): Color {
      if (btn == "C" || btn == "AC")
          return Color(0xFF908E92)
        if (btn == "(" || btn == ")")
            return Color(0xFF555457)
        if (btn == "-" || btn == "+" ||btn == "*" || btn == "/" || btn == "=")
            return Color(0xFF0762a2)
        return Color(0xFF3D414B)

    }
